#include <iostream.h>
    void selectionSortDescending(int arr[], int n) {
                    
        for (int i = 0; i < n - 1; i++) {
            int maxIndex = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] > arr[maxIndex]) {
                    maxIndex = j;
                }
            }int temp=arr[maxIndex];
            arr[maxIndex]=arr[i];
           arr[i]=temp; }
    }
    
    int main() {
        int arr[100];
        int n,i;
    
        cout << "Enter the number of elements: ";
        cin >> n;
    
    if (n>100){cout<<"error maximum size is 100."<> arr[i];
        
        }
    
        selectionSortDescending(arr,n);
    
        cout << "Sorted array in descending order: ";
        for (i = 0; i < n; i++) {
            cout << arr[i] << " ";
        }
        cout << endl;
    
        return 0;
    }
