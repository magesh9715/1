#include <iostream.h>

	void insertionSortDescending(int arr[], int n) {

		    for (int i = 1; i < n; i++) {
			int key = arr[i];
			int j = i - 1;
			while (j >= 0 && arr[j] < key) {
			    arr[j + 1] = arr[j];
			    j = j - 1;
			}
			arr[j + 1] = key;
		    }
		}

	int main() {
		    int arr[100];
		    int n,i;

		    cout << "Enter the number of elements: ";
		    cin >> n;
		    if (n>100){cout<<"error maximum size is 100."<> arr[i];
		    }

		    insertionSortDescending(arr,n);
		    cout << "Sorted array in descending order: ";

			cout << arr[i] << " ";
		    }
		    cout << endl;

		    return 0;
		}
