#include <iostream.h>
    struct Node {
        int data;
        Node *left, *right;
    
        Node(int val) {
            data = val;
            left = NULL;
            right = NULL;
        }
    };
    Node* insert(Node* root, int data) {
        if (root == NULL){
            return new Node(data);
        }
        if (data < root->data) {
            root->left = insert(root->left, data);
        } else {
            root->right = insert(root->right, data);
        }
        return root;
    }
    
    void inOrder(Node* root) {
        if (root != NULL){
            inOrder(root->left);
            cout << root->data << " ";
            inOrder(root->right);
        }
    }
    
    void preOrder(Node* root) {
        if (root != NULL){
            cout << root->data << " ";
            preOrder(root->left);
            preOrder(root->right);
        }
    }
    
    void postOrder(Node* root) {
        if (root != NULL){
            postOrder(root->left);
            postOrder(root->right);
            cout << root->data << " ";
        }
    }
    
    int main() {
        Node* root = NULL;
        int n, val;
    
        cout << "Enter the number of elements to insert in the BST: ";
        cin >> n;
    
        cout << "Enter the elements: ";
        for (int i = 0; i < n; i++) {
            cin >> val;
            root = insert(root, val);
        }
    
        cout << "In-order traversal: ";
        inOrder(root);
        cout << endl;
    
        cout << "Pre-order traversal: ";
        preOrder(root);
        cout << endl;
    
        cout << "Post-order traversal: ";
        postOrder(root);
        cout << endl;
    
        return 0;
    }
